project(libmathutils VERSION 1.0.0 LANGUAGES CXX)


# Declare your module

            src/matrix_tool.cpp
            src/integrate.cpp
            src/stochastic.cpp

)

    PUBLIC
        #$<INSTALL_INTERFACE:include>

    PRIVATE
        ${CMAKE_CURRENT_SOURCE_DIR}/src
)
